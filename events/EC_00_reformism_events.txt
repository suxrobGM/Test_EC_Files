# Популярность больше 10%
country_event = {
	id = reformism.1
	title = reformism.1.t
	desc = reformism.1.d
	picture = GFX_event_reformism_rise_10

	trigger = {
		reformism > 0.1
		NOT = {
			has_government = reformism
			has_country_flag = reformism_10
			has_idea = single-party_state
		}
	}
	mean_time_to_happen = {
		days = 30
		modifier = {
			has_stability > 49.9
			has_stability < 60
			factor = 1.2
		}
		modifier = {
			has_stability > 59.9
			has_stability < 70
			factor = 1.4
		}
		modifier = {
			has_stability > 69.9
			has_stability < 80
			factor = 1.6
		}
		modifier = {
			has_stability > 79.9
			has_stability < 90
			factor = 1.8
		}
		modifier = {
			has_stability > 89.9
			factor = 2
		}
	}

	immediate = {
		set_country_flag = reformism_10
	}

	option = { # Коалиция с реформистами
		name = reformism.1.a
		trigger = {
			OR = {
				has_government = communism
				has_government = liberalism
				has_government = democratic
				has_government = neutrality
				has_government = fascism
			}
			is_democracy = yes
		}
		ai_chance = {
			base = 5
			modifier = {
				factor = 55
				OR = {
					has_country_flag = pai_democrat_left
					has_country_flag = pai__reformist
				}
			}
			modifier = {
				factor = 0
				has_country_flag = pai_democrat_right
			}
		}
		random_list = {
			80 = {
				coalition_with_reformists = yes # Создать коалицию с реформистами
			}
			20 = {}
		}
		add_reformist = yes # Можно добавить реформиста в правительство
	}

	option = { # Не враждовать с реформистами
		name = reformism.1.b
		ai_chance = {
			base = 35
		}
		add_reformist = yes # Можно добавить реформиста в правительство
	}

	option = { # Выступать против реформизма
		name = reformism.1.c
		ai_chance = {
			base = 60
			modifier = {
				factor = 20
				has_country_flag = pai_nondem_neutral
			}
			modifier = {
				factor = 10
				has_country_flag = pai_democrat_neutral
			}
			modifier = {
				factor = 10
				has_country_flag = pai_democrat_left
			}
			modifier = {
				factor = 0
				has_country_flag = pai_nondem_reformator
			}
			modifier = {
				factor = 0
				has_country_flag = pai__communist
			}
		}
		set_country_flag = oppose_reformism
		add_popularity = {
			ideology = reformism
			popularity = -0.01
		}
		add_stability = -0.005
		add_political_power = -20
	}

	option = { # Подавлять реформизм
		name = reformism.1.s
		trigger = {
			can_suppress_ideologies = yes
		}
		ai_chance = {
			base = 50
			modifier = {
				factor = 0
				OR = {
					has_country_flag = pai_democrat_neutral
					has_country_flag = pai_democrat_left
					has_country_flag = pai_nondem_neutral
					has_country_flag = pai_nondem_reformator
					has_country_flag = pai__reformist
				}
			}
		}
		set_country_flag = suppress_reformism
		add_popularity = {
			ideology = reformism
			popularity = -0.03
		}
		add_stability = -0.01
		add_political_power = -30
	}
}
