add_namespace = kazakhstan

country_event = {  #Зашита от угрозы Советского Союза
	id = kazakhstan.1
	title = kazakhstan.1.t
	desc = kazakhstan.1.d
	picture = GFX_event_KAZ_army
	
	fire_only_once = yes

	trigger = { 
		tag = KAZ
		country_exists = KAZ
		KAZ = { has_country_flag = KAZ_deterence_agression_RUS_UNION }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = kazakhstan.1.a  
		create_faction = MASS
		add_to_faction = KAZ
		if = {
			limit = { 
				AND = {
					country_exists = UZB 
					UZB = { is_puppet = no }
				}
			}
			UZB = { country_event = { id = kazakhstan.2 days = 2 } }
		}
		if = {
			limit = { 
				AND = {
					country_exists = KGZ 
					KGZ = { is_puppet = no }
				}
			}
			KGZ = { country_event = { id = kazakhstan.2 days = 3 } }
		}
		if = {
			limit = { 
				AND = {
					country_exists = TJK 
					TJK = { is_puppet = no }
				}
			}
			TJK = { country_event = { id = kazakhstan.2 days = 4 } }
		}
		if = {
			limit = { 
				AND = {
					country_exists = TKM 
					TKM = { is_puppet = no }
				}
			}
			TKM = { country_event = { id = kazakhstan.2 days = 5 } }
		}
	}
}
