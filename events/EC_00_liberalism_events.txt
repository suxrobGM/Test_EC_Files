# Популярность больше 10%
country_event = {
	id = liberalism.1
	title = liberalism.1.t
	desc = liberalism.1.d
	picture = GFX_event_liberalism_rise_10

	trigger = {
		liberalism > 0.1
		NOT = {
			has_government = liberalism
			has_country_flag = liberalism_10
			has_idea = single-party_state
		}
	}
	mean_time_to_happen = {
		days = 30
		modifier = {
			has_stability > 49.9
			has_stability < 60
			factor = 1.2
		}
		modifier = {
			has_stability > 59.9
			has_stability < 70
			factor = 1.4
		}
		modifier = {
			has_stability > 69.9
			has_stability < 80
			factor = 1.6
		}
		modifier = {
			has_stability > 79.9
			has_stability < 90
			factor = 1.8
		}
		modifier = {
			has_stability > 89.9
			factor = 2
		}
	}

	immediate = {
		set_country_flag = liberalism_10
	}

	option = { # Коалиция с либералами
		name = liberalism.1.a
		trigger = {
			is_democracy = yes
			OR = {
				has_government = reformism
				has_government = democratic
				has_government = neutrality
				has_government = fascism
			}
		}
		ai_chance = {
			factor = 5
			modifier = {
				factor = 55
				OR = {
					has_country_flag = pai_democrat_right
					has_country_flag = pai__liberal
				}
			}
			modifier = {
				factor = 0
				has_country_flag = pai_democrat_left
			}
		}
		random_list = {
			80 = {
				coalition_with_liberals = yes # Создать коалицию с либералами
			}
			20 = {}
		}
		add_liberal = yes # Можно добавить либерала в правительство
	}

	option = { # Не враждовать с либералами
		name = liberalism.1.b
		ai_chance = {
			base = 35
		}
		add_liberal = yes # Можно добавить либерала в правительство
	}

	option = { # Выступать против либерализма
		name = liberalism.1.c
		ai_chance = {
			base = 60
			modifier = {
				factor = 20
				has_country_flag = pai_nondem_neutral
			}
			modifier = {
				factor = 20
				has_country_flag = pai_democrat_right
			}
			modifier = {
				factor = 10
				has_country_flag = pai_democrat_neutral
			}
			modifier = {
				factor = 0
				has_country_flag = pai__liberal
			}
		}
		set_country_flag = oppose_liberalism
		add_popularity = {
			ideology = liberalism
			popularity = -0.01
		}
		add_stability = -0.005
		add_political_power = -20
	}

	option = { # Подавлять либерализм
		name = liberalism.1.s
		trigger = {
			can_suppress_ideologies = yes
		}
		ai_chance = {
			base = 50
			modifier = {
				factor = 10
				has_country_flag = pai_democrat_right
			}
			modifier = {
				factor = 0
				OR = {
					has_country_flag = pai_democrat_neutral
					has_country_flag = pai_nondem_neutral
					has_country_flag = pai__liberal
				}
			}
		}
		set_country_flag = suppress_liberalism
		add_popularity = {
			ideology = liberalism
			popularity = -0.03
		}
		add_stability = -0.01
		add_political_power = -30
	}
}
