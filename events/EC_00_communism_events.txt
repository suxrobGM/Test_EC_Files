# Популярность больше 10%
country_event = {
	id = communism.1
	title = communism.1.t
	desc = communism.1.d
	picture = GFX_event_communism_rise_10

	trigger = {
		communism > 0.1
		NOT = {
			has_government = communism
			has_country_flag = communism_10
			has_idea = single-party_state
		}
	}
	mean_time_to_happen = {
		days = 30
		modifier = {
			has_stability > 49.9
			has_stability < 60
			factor = 1.2
		}
		modifier = {
			has_stability > 59.9
			has_stability < 70
			factor = 1.4
		}
		modifier = {
			has_stability > 69.9
			has_stability < 80
			factor = 1.6
		}
		modifier = {
			has_stability > 79.9
			has_stability < 90
			factor = 1.8
		}
		modifier = {
			has_stability > 89.9
			factor = 2
		}
	}

	immediate = {
		set_country_flag = communism_10
	}

	option = { # Коалиция с коммунистами
		name = communism.1.a
		trigger = {
			has_government = reformism
			is_democracy = yes
		}
		ai_chance = {
			base = 5
			modifier = {
				factor = 30
				has_country_flag = pai_democrat_left
			}
			modifier = {
				factor = 20
				has_country_flag = pai_nondem_reformator
			}
			modifier = {
				factor = 50
				has_country_flag = pai__communist
			}
			modifier = {
				factor = 0
				OR = {
					has_country_flag = pai_democrat_anticommunist
					has_country_flag = pai_democrat_moderate
					has_country_flag = pai_democrat_right
					has_country_flag = pai_nondem_totalitarian
				}
			}
		}
		random_list = {
			80 = {
				coalition_with_communists = yes # Создать коалицию с коммунистами
			}
			20 = {}
		}
		add_communist = yes # Можно добавить коммуниста в правительство
	}

	option = { # Не враждовать с коммунистами
		name = communism.1.b
		add_communist = yes # Можно добавить коммуниста в правительство
		ai_chance = {
			base = 35
			modifier = {
				factor = 75
				has_country_flag = pai_democrat_neutral
			}
			modifier = {
				factor = 60
				has_country_flag = pai_democrat_left
			}
			modifier = {
				factor = 80
				has_country_flag = pai_nondem_reformator
			}
			modifier = {
				factor = 50
				has_country_flag = pai__communist
			}
			modifier = {
				factor = 0
				OR = {
					has_country_flag = pai_democrat_anticommunist
					has_country_flag = pai_democrat_right
					has_country_flag = pai_nondem_totalitarian
				}
			}
		}
	}

	option = { # Выступать против коммунизма
		name = communism.1.c
		ai_chance = {
			base = 60
			modifier = {
				factor = 20
				has_country_flag = pai_democrat_neutral
			}
			modifier = {
				factor = 10
				has_country_flag = pai_democrat_left
			}
			modifier = {
				factor = 0
				has_country_flag = pai__communist
			}
		}		
		set_country_flag = oppose_communism
		add_popularity = {
			ideology = communism
			popularity = -0.01
		}
		add_stability = -0.005
		add_political_power = -20
	}

	option = { # Подавлять коммунизм
		name = communism.1.s
		trigger = {
			can_suppress_ideologies = yes
		}
		ai_chance = {
			base = 50
			modifier = {
				factor = 100
				has_country_flag = pai_democrat_anticommunist
			}
			modifier = {
				factor = 0
				OR = {
					has_country_flag = pai_democrat_neutral
					has_country_flag = pai_democrat_antinationalist
					has_country_flag = pai_democrat_left
					has_country_flag = pai__communist
				}
			}
		}
		set_country_flag = suppress_communism
		add_popularity = {
			ideology = communism
			popularity = -0.03
		}
		add_stability = -0.01
		add_political_power = -30
	}
}
