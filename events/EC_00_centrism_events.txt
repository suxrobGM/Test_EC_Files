add_namespace = centrism

# Популярность больше 10%
country_event = {
	id = centrism.1
	title = centrism.1.t
	desc = centrism.1.d
	picture = GFX_event_centrism_rise_10

	trigger = {
		neutrality > 0.1
		NOT = {
			has_government = neutrality
			has_country_flag = centrism_10
			has_idea = single-party_state
		}
	}
	mean_time_to_happen = {
		days = 30
		modifier = {
			has_stability > 49.9
			has_stability < 60
			factor = 1.2
		}
		modifier = {
			has_stability > 59.9
			has_stability < 70
			factor = 1.4
		}
		modifier = {
			has_stability > 69.9
			has_stability < 80
			factor = 1.6
		}
		modifier = {
			has_stability > 79.9
			has_stability < 90
			factor = 1.8
		}
		modifier = {
			has_stability > 89.9
			factor = 2
		}
	}

	immediate = {
		set_country_flag = centrism_10
	}

	option = { # Коалиция с центристами
		name = centrism.1.a
		trigger = {
			is_democracy = yes
			OR = {
				has_government = reformism
				has_government = liberalism
				has_government = democratic
				has_government = fascism
				has_government = islamists
			}
		}
		ai_chance = {
			base = 5
			modifier = {
				factor = 65
				has_country_flag = pai__centrist
			}
			modifier = {
				factor = 0
				has_country_flag = pai_nondem_totalitarian
			}
		}
		random_list = {
			80 = {
				coalition_with_centrists = yes # Создать коалицию с центристами
			}
			20 = {}
		}
		add_centrist = yes # Можно добавить центриста в правительство
	}

	option = { # Не враждовать с центристами
		name = centrism.1.b
		ai_chance = {
			base = 35
		}
		add_centrist = yes # Можно добавить центриста в правительство
	}

	option = { # Выступать против центризма
		name = centrism.1.c
		ai_chance = {
			base = 60
			modifier = {
				factor = 20
				has_country_flag = pai_nondem_neutral
			}
			modifier = {
				factor = 10
				has_country_flag = pai_democrat_neutral
			}
			modifier = {
				factor = 0
				has_country_flag = pai__centrist
			}
		}
		set_country_flag = oppose_centrism
		add_popularity = {
			ideology = neutrality
			popularity = -0.01
		}
		add_stability = -0.005
		add_political_power = -20
	}

	option = { # Подавлять центризм
		name = centrism.1.s
		trigger = {
			can_suppress_ideologies = yes
		}
		ai_chance = {
			base = 50
			modifier = {
				factor = 0
				OR = {
					has_country_flag = pai_democrat_neutral
					has_country_flag = pai_nondem_neutral
					has_country_flag = pai__centrist
				}
			}
		}
		set_country_flag = suppress_centrism
		add_popularity = {
			ideology = neutrality
			popularity = -0.03
		}
		add_stability = -0.01
		add_political_power = -30
	}
