# Популярность больше 10%
country_event = {
	id = islamism.1
	title = islamism.1.t
	desc = islamism.1.d
	picture = GFX_event_islamism_rise_10

	trigger = {
		islamists > 0.1
		NOT = {
			has_government = islamists
			has_country_flag = islamism_10
			has_idea = single-party_state
		}
	}
	mean_time_to_happen = {
		days = 30
		modifier = {
			has_stability > 49.9
			has_stability < 60
			factor = 1.2
		}
		modifier = {
			has_stability > 59.9
			has_stability < 70
			factor = 1.4
		}
		modifier = {
			has_stability > 69.9
			has_stability < 80
			factor = 1.6
		}
		modifier = {
			has_stability > 79.9
			has_stability < 90
			factor = 1.8
		}
		modifier = {
			has_stability > 89.9
			factor = 2
		}
	}

	immediate = {
		set_country_flag = islamism_10
	}

	option = { # Коалиция с исламистами
		name = islamism.1.a
		trigger = {
			is_democracy = yes
			OR = {
				has_government = neutrality
				has_government = fascism
			}
		}
		ai_chance = {
			base = 5
			modifier = {
				factor = 65
				has_country_flag = pai__islamist
			}
			modifier = {
				factor = 0
				OR = {
					has_country_flag = pai_nondem_totalitarian
					has_country_flag = pai_democrat_moderate
				}
			}
		}
		random_list = {
			80 = {
				coalition_with_islamists = yes # Создать коалицию с исламистами
			}
			20 = {}
		}
		add_islamist = yes # Можно добавить исламиста в правительство
	}

	option = { # Не враждовать с исламистами
		name = islamism.1.b
		ai_chance = {
			base = 35
			modifier = {
				factor = 0
				has_country_flag = pai_democrat_moderate
			}
		}
		add_islamist = yes # Можно добавить исламиста в правительство
	}

	option = { # Выступать против исламизма
		name = islamism.1.c
		ai_chance = {
			base = 60
			modifier = {
				factor = 20
				has_country_flag = pai_nondem_neutral
			}
			modifier = {
				factor = 10
				has_country_flag = pai_democrat_neutral
			}
			modifier = {
				factor = 0
				has_country_flag = pai__islamist
			}
		}
		set_country_flag = oppose_islamism
		add_popularity = {
			ideology = islamists
			popularity = -0.01
		}
		add_stability = -0.005
		add_political_power = -20
	}

	option = { # Подавлять исламизм
		name = islamism.1.s
		trigger = {
			can_suppress_ideologies = yes
		}
		ai_chance = {
			base = 50
			modifier = {
				factor = 0
				OR = {
					has_country_flag = pai_democrat_neutral
					has_country_flag = pai_nondem_neutral
					has_country_flag = pai__islamist
				}
			}
		}
		set_country_flag = suppress_islamism
		add_popularity = {
			ideology = islamists
			popularity = -0.03
		}
		add_stability = -0.01
		add_political_power = -30
	}
}
